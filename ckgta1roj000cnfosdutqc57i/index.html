ElasticSearch之ScrollAPI

发表于 2020-10-16 更新于 2020-10-21 分类于 es 阅读次数：

    最近遇到了一个需求，公司集团的所有项目日志都是存在kafka中，通过logstash去不断读取，通过一定的规则将日志切割为我们需要的格式，存入ES。我们要做的就是定期不断的从ES中取获取日志，分析日志。

  1. Scroll的概念

    Scroll其实和文件读取中的指针(seek)，传统数据库中的游标非常相似，目的是为了在数据量特别大的时候，去检索数据。因为不可能在一次请求中将所有数据一并处理。

  2. 基本使用

    ES检索数据所使用的api使用GET,POST方法都是可以的，query scroll 的意思是表明此次会话的有效时长是多久，如果过期的话ES会自动销毁，不过在实际使用过程中，最好还是使用完毕后主动销毁,因为保存会话的存活是有成本的。DSL查询条件则存放于body中

    • 首次查询

      POST /index/_search?scroll=1m
{
  "size":1000,
  "query":{
    "range":{
      "timestamp":{
        "gt":"xxxxx",  # 我这里是检索一定时间范围的数据，注意，这里的时间是UTC格式的时间字符串
        "lt":"xxxxx",
      }
    }
  }
}

    • 在此次请求结果中，除了会得到正常的search response外，还会得到一个 _scroll_id ，而我们后续的请求就会使用这个id去请求scroll api，直到hits中的数据长度为0.

      POST /_search/scroll
# 注意 scroll 请求url中不能包含 index 和 type，因为初始请求已经包含。（type 在es7以后已经不再推荐使用了）
{
  "scroll": "1m", # 让此次会话再保存1分钟
  "scroll_id": [_scroll_id] # 首次请求所的到的id
}

    • 销毁会话 可以在query中传入scroll_id 也可以在body中

      DELETE /_search/id1,id2
{
  "scroll_id": "xxxxx",
  # 也可以使用数组传递多个
  "scroll_id": [
    "id1",
    "id2"
  ]
}

# 也可以一次性删除所有
DELETE /_search/scroll/_all

  3. 完成需求代码

    import (
  "logs"
  "context"
  "fmt"
  requests "github.com/levigross/grequests"
  "time"
)

type (
  searchBody struct {
    Query _query `json:"query"`
    Size  int    `json:"size"`
  }

  _query struct {
    Range _range `json:"range"`
  }

  _range struct {
    Timestamp _timestamp `json:"timestamp"`
  }

  _timestamp struct {
    Gte string `json:"gte"`
    Lte string `json:"lte"`
  }

  searchResult struct {
    ScrollId string `json:"_scroll_id"`
    Hits     _hits  `json:"hits"`
  }

  _hits struct {
    Hits []map[string]interface{} `json:"hits"`
  }
)

func job(ctx context.Context) {
  var (
    err   error
    index string
    resp  *requests.Response
  )

  const (
    utcFormat  = "2006-01-02T15:04:05Z"
    searchSize = 1000
    scroll     = "10m"
  )

  now := time.Now()
  start := now.Add(-time.Hour)

  gte := start.Format(utcFormat)
  lte := now.Format(utcFormat)

  body := searchBody{
    Query: _query{_range{_timestamp{
      Gte: gte,
      Lte: lte,
    }}},
    Size: searchSize,
  }

  defer func() {
    _ = resp.Close()
  }()

  if resp, err = requests.Get(fmt.Sprintf("http://%s/%s/_search?scroll=%s", esHost, index, scroll),
    &requests.RequestOptions{
      Context: ctx,
      Auth:    []string{esUser, esPass},
      JSON:    body,
    }); err != nil {
    logs.Errorf(err.Error())
    return
  }

  var (
    serRes searchResult
    tmp    []map[string]interface{}
  )

  if err = resp.JSON(&serRes); err != nil {
    logs.Errorf("解析json失败:%s", err.Error())
    return
  }

  tmp = append(tmp, serRes.Hits.Hits...)

  defer func() {
    // 清除会话
    if dResp, err := requests.Delete(fmt.Sprintf("http://%s/_search/scroll/%s", esHost, serRes.ScrollId),
      &requests.RequestOptions{Auth: []string{esUser, esPass}}); err != nil {
      logs.Errorf("清楚scroll会话失败:%s", err.Error())
    } else {
      _ = dResp.Close()
    }
  }()

  for {
    var er searchResult
    if resp, err = requests.Get(fmt.Sprintf("http://%s/_search/scroll", esHost),
      &requests.RequestOptions{
        Context: ctx,
        Auth:    []string{esUser, esPass},
        JSON: map[string]interface{}{
          "scroll":    scroll,
          "scroll_id": serRes.ScrollId,
        },
      }); err != nil {
      logs.Errorf("scroll error : %s", err.Error())
      return
    }

    if err = resp.JSON(&er); err != nil {
      logs.Errorf("scroll json error : %s", err.Error())
      return
    }

    if len(er.Hits.Hits) == 0 {
      break
    }

    tmp = append(tmp, er.Hits.Hits...)
  }

  logs.Infof("最终结果:%s", tmp)
}
