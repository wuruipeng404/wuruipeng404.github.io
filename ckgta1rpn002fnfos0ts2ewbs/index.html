理解设计模式之责任链模式

发表于 2020-10-27 分类于 设计模式 阅读次数：

责任链模式：分离不同职责 动态的组合相关责任链
责任链上的处理者负责处理请求。客户端只需要发送即可
这里举一个 通俗易懂的例子，叫家人帮我买手机！

package main

import "fmt"

// 定义一个购买手机的需求
type Require struct {
	phone string
	price int
}

// 定义处理请求的接口
type Handler interface {
	ProcessRequire(require *Require)
}

// 定义一系列可以帮我买手机的家人们
// 当前的处理者处理不了请求,就要传给下一位,因此每位处理者需要保存下一位处理者的指针
type (
	Father struct {
		next Handler
	}

	Mother struct {
		next Handler
	}

	GrandFather struct {
		next Handler
	}

	GrandMother struct {
	}
)

func (h *Father) ProcessRequire(require *Require) {

	if require == nil {
		return
	}

	if require.price <= 500 {
		fmt.Println("爸爸帮你买了!")
	} else {
		h.next.ProcessRequire(require)
		fmt.Println("爸爸买不起交给妈妈")
	}
}

// 略
