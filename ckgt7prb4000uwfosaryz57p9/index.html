理解设计模式之原型模式

发表于 2020-10-23 更新于 2020-10-28 分类于 设计模式 阅读次数：

原型模式和其他创建型模式不一样的是，它是复制一个已有对象，而不是创建一个新对象，被复制的对象就叫做原型。
原型模式多用于建造复杂或者耗时的对象，这样直接复制会比较高效。
原型对象一般会配合原型管理器一起使用

  1. go

    package main

import (
	"fmt"
	"reflect"
)

type Cloneable interface {
	Clone() Cloneable
}

// 原型管理器
type ProtoManager struct {
	prototypes map[string]Cloneable
}

func NewProtoManager() *ProtoManager {
	return &ProtoManager{prototypes: make(map[string]Cloneable)}
}

func (p *ProtoManager) Get(name string) Cloneable {
	return p.prototypes[name]
}
func (p *ProtoManager) Set(name string, v Cloneable) {
	p.prototypes[name] = v
}

type Object struct {
	name string
}

func (a *Object) Clone() Cloneable {
	tc := *a
	return &tc
}

func NewObject(name string) *Object {
	return &Object{name: name}
}

func main() {
	manager := NewProtoManager()
	a := NewObject("rumple")
	manager.Set("t1", a)

	c := manager.Get("t1").Clone()
	fmt.Println(c)

	// 判断是否为同一对象
	fmt.Println(a == c)
	// 判断是否相等
	fmt.Println(reflect.DeepEqual(a, c))
}

  2. python

    import abc

import copy


class ProtoType(metaclass=abc.ABCMeta):
    """
    原型抽象类
    """

    @abc.abstractmethod
    def clone(self):
        pass

    @abc.abstractmethod
    def deep_clone(self):
        pass


class WorkExperience(object):
    """
    工作经验类
    """

    def __init__(self):
        self.work_time = None
        self.company = None

    def set_work(self, time, company: str):
        self.work_time = time
        self.company = company


class Resume(ProtoType):
    """
    简历类
    """

    def __init__(self, name: str):
        self.name = name
        self.work_experience = WorkExperience()
        self.gender = None
        self.age = None

    def set_person_info(self, gender: str, age: int):
        self.gender = gender
        self.age = age

    def set_work_experience(self, time, company: str):
        self.work_experience.set_work(time, company)

    def display(self):
        print(self.name, self.gender, self.age)
        print(f"工作经历 {self.work_experience.company, self.work_experience.work_time}")

    def clone(self):
        return copy.copy(self)

    def deep_clone(self):
        return copy.deepcopy(self)


if __name__ == '__main__':
    a = Resume("rumple")
    b = a.clone()
    c = a.deep_clone()

    a.set_person_info("男", 67)
    a.set_work_experience('2018', "腾讯")

    b.set_person_info("女", 45)
    b.set_work_experience('2019', "阿里")

    c.set_person_info("女", 32)
    c.set_work_experience('2020', "百度")

    a.display()
    print("---------------------------------------")
    b.display()
    print("-----------------")
    c.display()
