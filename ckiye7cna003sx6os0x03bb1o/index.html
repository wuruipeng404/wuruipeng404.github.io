理解设计模式之观察者模式

发表于 2020-10-26 更新于 2020-12-21 分类于 设计模式 阅读次数：

观察者模式比较好理解，就是字面意思，其实和发布订阅差不多。
举一个学生订阅课程的例子

package main

// 定义课程接口
type ISubject interface {
	Attach(IObserver) // 添加观察者方法
	Notify()          // 通知接口
	UpdateTime(time string)
}

// 定义课程类
type Subject struct {
	observers []IObserver
	name      string // 课程名称
	time      string // 上课时间
}

func NewSubject(name string) ISubject {
	return &Subject{
		observers: make([]IObserver, 0),
		name:      name,
	}
}

// 添加观察者
func (s *Subject) Attach(o IObserver) {
	s.observers = append(s.observers, o)
}

// 通知观察者进行改变
func (s *Subject) Notify() {
	for _, o := range s.observers {
		o.Update(s)
	}
}

func (s *Subject) UpdateTime(time string) {
	s.time = time
}

// 定义观察者类
type IObserver interface {
	Update(subject ISubject)
}

type Student struct {
	name string
	sub  ISubject
}

func (s *Student) Update(subject ISubject) {
	s.sub = subject
}

func main() {
	sbj := NewSubject("线性代数")
	lily := &Student{name: "lily"}

	sbj.Attach(lily)
	sbj.UpdateTime("12点半")
	// ...
}
