impl Rumple {
    fn new() -> Rumple {
        Rumple {
            name: "rumple".to_string(),
            location: "Shanghai".to_string(),
            age: 28,
            hobby: vec![
                "Basketball".to_string(),
                "Swim".to_string(),
                "Read".to_string(),
                "Code".to_string(),
                "Games".to_string(),
            ],
            constellation: "Gemini".to_string(),
            mail: "d3JwMzU3NzExNTg5QDE2My5jb20".to_string(),
            motto: "All magic comes with a price".to_string(),
        }
    }

    fn eat(&self) {
        println!("{} eating", self.name);
    }

    fn code(&self) {
        println!("{} coding", self.name);
    }

    fn sleep(&self) {
        println!("{} sleeping", self.name);
    }

    fn life(&self) {
        loop {
            self.eat();
            self.code();
            self.sleep();
        }
    }
}


fn main() {
    Rumple::new().life()
}

	fmt.Println(r.Name, " eating")
}

func (r *Rumple) Sleep() {
	fmt.Println(r.Name, " sleeping")
}

func (r *Rumple) Code() {
	fmt.Println(r.Name, " coding")
}

func (r *Rumple) Life() {
	alive := true

	for alive {
		r.Eat()
		r.Code()
		r.Sleep()
	}

}

	NewRumple().Life()
